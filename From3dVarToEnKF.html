def Lorenz63(state,*args):       #Lorenz63
    sigma = args[0]
    beta = args[1]
    rho = args[2]
    x, y, z = state 
    f = np.zeros(3) 
    f[0] = sigma * (y - x)
    f[1] = x * (rho - z) - y
    f[2] = x * y - beta * z
    return f 
def RK4(rhs,state,dt,*args):    # Runge-Kuttarhs
    k1 = rhs(state,*args)
    k2 = rhs(state+k1*dt/2,*args)
    k3 = rhs(state+k2*dt/2,*args)
    k4 = rhs(state+k3*dt,*args) # 
    new_state = state + (dt/6)*(k1+2*k2+2*k3+k4)
    return new_state

def JLorenz63(state,*args):         # Lorenz63 
    sigma = args[0]
    beta = args[1]
    rho = args[2]
    x, y, z = state
    
    df = np.zeros([3,3])           # 9
    df[0,0] = sigma * (-1)
    df[0,1] = sigma * (1)
    df[0,2] = sigma * (0)
    df[1,0] = 1 * (rho - z) 
    df[1,1] = -1
    df[1,2] = x * (-1)
    df[2,0] = 1 * y 
    df[2,1] = x * 1 
    df[2,2] = - beta
    return df 

def JRK4(rhs,Jrhs,state,dt,*args):   #     
    n = len(state)
    k1 = rhs(state,*args)
    k2 = rhs(state+k1*dt/2,*args)
    k3 = rhs(state+k2*dt/2,*args)
    # Runge-Kutta
    dk1 = Jrhs(state,*args)
    dk2 = Jrhs(state+k1*dt/2,*args) @ (np.eye(n)+dk1*dt/2) 
    dk3 = Jrhs(state+k2*dt/2,*args) @ (np.eye(n)+dk2*dt/2) 
    dk4 = Jrhs(state+k3*dt,*args) @ (np.eye(n)+dk3*dt)  
    DM = np.eye(n) + (dt/6) * (dk1+2*dk2+2*dk3+dk4)
    return DM

def EnSRF(xbi,yo,ObsOp,Dh,R):
    from scipy.linalg import sqrtm
    n,N = xbi.shape                      # n-N-
    m = yo.shape[0]                      # m-
    xb = np.mean(xbi,1)                   #  
    B = (1/(N-1)) * (xbi - xb.reshape(-1,1)) @ (xbi - xb.reshape(-1,1)).T
    D = Dh@B@Dh.T + R
    K = B @ Dh.T @ np.linalg.inv(D)         # !!!EnKF
    xa = xb + K @ (yo-ObsOp(xb))          # 

    A = xbi - xb.reshape(-1,1)              # 
    Z = A/np.sqrt(N-1)                    # 
    Y = np.linalg.inv(D)@Dh@Z           
    X = sqrtm(np.eye(N)-(Dh@Z).T@Y)      # 
    X = np.real(X)                         # 
    
    Z = Z@X                             # 
    A = Z*np.sqrt(N-1)
    xai = xa.reshape(-1,1)+A               # 
    return xai

    n,N = xbi.shape         # n-N-
    m = yo.shape[0]         # m-
    xb = np.mean(xbi,1)     #  
    B = (1/(N-1)) * (xbi - xb.reshape(-1,1)) @ (xbi - xb.reshape(-1,1)).T
    D = Dh@B@Dh.T + R
    K = B @ Dh.T @ np.linalg.inv(D)     # !!!EnKF
    xa = xb + K @ (yo-ObsOp(xb))        # 
    
    A = xbi - xb.reshape(-1,1)          # 
    A = A - (1/2) * K @ Dh @ A          # !! DEnKF
    xai = xa.reshape(-1,1) + A
          
    return xai

def obs_increment_eakf(ensemble, observation, obs_error_var): # 1
    prior_mean = np.mean(ensemble);
    prior_var = np.var(ensemble);
    if prior_var >1e-6:                                                 # 
        post_var = 1.0 / (1.0 / prior_var + 1.0 / obs_error_var);       # 2
        post_mean = post_var * (prior_mean / prior_var + observation / obs_error_var);  # 3
    else:
        post_var = prior_var; post_mean = prior_mean;    
    updated_ensemble = ensemble - prior_mean + post_mean;
    var_ratio = post_var / prior_var;
    updated_ensemble = np.sqrt(var_ratio) * (updated_ensemble - post_mean) + post_mean; # 4
    obs_increments = updated_ensemble - ensemble;
    return obs_increments
def get_state_increments(state_ens, obs_ens, obs_incs): # 2
    covar = np.cov(state_ens, obs_ens);
    state_incs = obs_incs * covar[0,1]/covar[1,1];
    return state_incs
def sEAKF(xai,yo,ObsOp, R, RhoM):
    n,N = xai.shape;                  #
